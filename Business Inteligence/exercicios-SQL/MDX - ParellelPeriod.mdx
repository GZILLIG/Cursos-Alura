AS
'([Measures].[Faturamento], ParallelPeriod([Tempo].[Hierarquia do Mês e Ano].[Nivel Mês e Ano], 6, [Tempo].[Hierarquia do Mês e Ano].Currentmember))', 
// A ParallelPeriod desloca o resultado X numeros de tuplas à frente (nesse caso, 6)

({[Measures].[Faturamento], [Measures].[Faturamento Desce 2]}) On Columns
FROM [COMPLETO]

AS '([Measures].[Faturamento])/
([Measures].[Faturamento], ParallelPeriod([Tempo].[Hierarquia do Mês e Ano].[Nivel Mês e Ano], 12, 
[Tempo].[Hierarquia do Mês e Ano].Currentmember)) -1',
FORMAT_STRING = "#,###.00 %"

({[Measures].[Faturamento], [Measures].[Variação Faturamento Ano]}) On Columns
FROM [COMPLETO]
